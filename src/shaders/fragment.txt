#version 330 core

in vec3 fragColor;
in vec2 fragTexCoord;

out vec4 screenColor;

uniform sampler2D material; //0
uniform sampler2D mask;     //1

void main() {
    vec3 baseColor = texture(material, fragTexCoord).rgb;
    float alpha = texture(mask, fragTexCoord).r;
    screenColor = vec4(baseColor, alpha);
}
